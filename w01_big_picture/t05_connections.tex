	\begin{frame}[c]{Building a Successful AutoML Tool}

	 \begin{itemize}
	 	\item Hyperparameter optimization
	 	\item Neural architecture search
	 	\item Model and algorithm selection
	 	\item Dynamic approaches
	 \end{itemize}
